async function fetchAndParseMetadata(url) { try { const response = await fetch(url); const html = await response.text(); const tempDiv = document.createElement('div'); tempDiv.innerHTML = html; const metadata = {}; const titleElement = tempDiv.querySelector('title'); metadata.title = titleElement ? titleElement.textContent.trim() : ''; const descriptionElement = tempDiv.querySelector('meta[name="description"]'); metadata.description = descriptionElement ? descriptionElement.getAttribute('content') : ''; const keywordsElement = tempDiv.querySelector('meta[name="keywords"]'); metadata.keywords = keywordsElement ? keywordsElement.getAttribute('content').split(',') : []; return metadata; } catch (error) { console.error('Error fetching or parsing metadata:', error); return null; } } async function fetchAndDisplayMetadataForUrls() { const urlList = document.getElementById('urlList').querySelectorAll('li'); const metadataContainer = document.getElementById('metadataContainer'); metadataContainer.innerHTML = ''; // Clear previous metadata for (const urlListItem of urlList) { const url = urlListItem.textContent.trim(); try { const metadata = await fetchAndParseMetadata(url); if (metadata) { const metadataDiv = document.createElement('div'); metadataDiv.innerHTML = `

URL: ${url}

Title: ${metadata.title}

Description: ${metadata.description}

Keywords: ${metadata.keywords.join(', ')}

`; metadataContainer.appendChild(metadataDiv); } else { const errorDiv = document.createElement('div'); errorDiv.textContent = `Failed to fetch metadata for ${url}`; metadataContainer.appendChild(errorDiv); } } catch (error) { console.error('Error fetching metadata for', url, ':', error); const errorDiv = document.createElement('div'); errorDiv.textContent = `Error fetching metadata for ${url}`; metadataContainer.appendChild(errorDiv); } } } fetchAndDisplayMetadataForUrls();
